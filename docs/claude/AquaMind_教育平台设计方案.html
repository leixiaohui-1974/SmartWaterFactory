## 软测量流程（学生用Jupyter一步步实现） # 1. 特征工程 X = df[['DO', 'pH', 'EC', 'NTU', 'T', 'ORP', 'Q_air', 'dose_PAC']] y_COD = df['COD_lab'] # 人工化验值（每周测一次） # 2. 时序特征扩展（滚动窗口均值） X['DO_ma6h'] = X['DO'].rolling('6h').mean() X['NTU_diff'] = X['NTU'].diff(periods=12) # 3. XGBoost基础模型 from xgboost import XGBRegressor model_cod = XGBRegressor(n_estimators=200, max_depth=5) model_cod.fit(X_train, y_COD_train) # 4. LSTM时序残差修正（捕捉动态规律） # 输入：过去24步特征序列 输出：XGBoost残差修正量 lstm_residual = LSTM(input_dim=8, hidden_dim=64, seq_len=24) y_final = model_cod.predict(X) + lstm_residual.predict(X_seq) # 目标：RMSE(COD) < 10 mg/L，RMSE(NH4) < 2 mg/L
② DO级联控制 + 节能间歇曝气（Year 2~3）
Year 2+
## 自适应间歇曝气策略（比连续曝气节能 20~35%） class AdaptiveAeration: def __init__(self, DO_target=2.0, min_aer=120, min_mix=60): self.DO_sp = DO_target self.pid = PID(Kp=0.8, Ki=0.05, Kd=0.1) def update(self, DO_meas, NH4_load): # 外环：DO设定点随氨氮负荷动态调整 self.DO_sp = 1.5 + 0.8 * (NH4_load / NH4_ref) # 内环：PID → MFC指令 Q_air = self.pid.compute(self.DO_sp, DO_meas) # 节能判断：DO充足时切换搅拌不曝气 if DO_meas > self.DO_sp + 0.3: return {'mode': 'mix_only', 'Q_air': 0} else: return {'mode': 'aerate', 'Q_air': clip(Q_air, 0.5, 5.0)}

## do-mpc多变量MPC（控制DO + 投药量，预测水质达标） import do_mpc model = do_mpc.model.Model('continuous') # 状态：[COD, NH4, DO, MLSS] 控制：[Q_air, D_PAC, D_NaOH] x = model.set_variable('_x', 'states', (4,1)) u = model.set_variable('_u', 'controls', (3,1)) # 右端项：ASM1简化 + LSTM残差 model.set_rhs('states', ASM1_simplified(x, u) + LSTM_correction(x_hist)) # MPC目标：最小化水质偏差 + 能耗 mterm = (NH4_out - NH4_target)**2 * 100 lterm = mterm + w_e * Q_air**2 + w_d * D_PAC**2 # 约束：水质合规 + 执行器限制 mpc.bounds['lower', '_u', 'Q_air'] = 0.5 # L/min mpc.bounds['upper', '_u', 'Q_air'] = 8.0 # Np=12步(12min), Nc=3步控制域, Ts=60s # → 求解时间 < 200ms（Jetson Orin上）
